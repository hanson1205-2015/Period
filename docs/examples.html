Examples

Factorial

define factorial with n.
    if n <= 1 then.
        return 1.
    end if.
    return n * factorial with n - 1.
end define.

show factorial with 5.

Fibonacci

define fib with n.
    if n <= 1 then.
        return n.
    end if.
    return fib with n - 1 + fib with n - 2.
end define.

let i be 0.
while i < 10 repeat.
    show fib with i.
    set i to i + 1.
end while.

Read and Greet

show "What is your name?".
let name be input.
show "Nice to meet you, " + name + ".".
